        // Chatbot'u Aç/Kapat
        function toggleChatbot() {
            const chatbotContainer = document.getElementById("chatbot-container");
            chatbotContainer.classList.toggle("active");
        }

        // Mesaj Gönderme Fonksiyonu
        function sendMessage(userInput) {
            const chatbotMessages = document.getElementById("chatbot-messages");

            // Kullanıcı Mesajını Ekle
            if (userInput) {
                chatbotMessages.innerHTML += `
                    <div class="message user">
                        <div class="message-content">
                            ${userInput}
                        </div>
                    </div>
                `;
            }

            // Chatbot Yanıtını Oluştur
            let botResponse = botResponses.default;
            for (const key in botResponses) {
                if (userInput.includes(key)) {
                    botResponse = botResponses[key];
                    break;
                }
            }

            // Kargo Takip İsteği
            if (userInput.includes("kargo")) {
                document.getElementById("tracking-input").style.display = "block";
            } else {
                document.getElementById("tracking-input").style.display = "none";
            }

            // Yazma Animasyonu
            chatbotMessages.innerHTML += `
                <div class="message bot">
                    <div class="message-content">
                        <span class="typing-indicator">
                            <span></span>
                            <span></span>
                            <span></span>
                        </span>
                    </div>
                </div>
            `;

            // Yazma Animasyonu Sonrası Yanıtı Göster
            setTimeout(() => {
                document.querySelector(".typing-indicator").remove();
                chatbotMessages.innerHTML += `
                    <div class="message bot">
                        <div class="message-content">
                            ${botResponse}
                        </div>
                    </div>
                `;
                // Mesajlar alanını en altına kaydır
                chatbotMessages.scrollTop = chatbotMessages.scrollHeight;
            }, 1500);

            // Input Alanını Temizle
            document.getElementById("user-input").value = "";
        }

            }
        }

        function showCategories() {
            const categoriesView = document.getElementById("categories-view");
            categoriesView.style.display = categoriesView.style.display === "block" ? "none" : "block";

            // Kategorileri dinamik olarak doldurma
            const categories = ["Telefonlar", "Tabletler", "Aksesuarlar", "Kampanyalar"];
            const categoryList = document.getElementById("category-list");
            categoryList.innerHTML = "";
            categories.forEach(category => {
                const categoryItem = document.createElement("div");
                categoryItem.className = "category-item";
                categoryItem.textContent = category;
                categoryItem.addEventListener("click", () => {
                    sendMessage(category);
                });
                categoryList.appendChild(categoryItem);
            });
        }
